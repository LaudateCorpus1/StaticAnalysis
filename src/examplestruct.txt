struct Call_t{
	std::list<duint> arguments;             // address of arguments
	duint RVA;                              // relative VA of call
	duint RelativeAddressValue;             // call "xxxx"
	bool isApiCall;


	void clear(){
		arguments.clear();
		RVA = 0;
		RelativeAddressValue = 0;
		isApiCall = false;
	}
	Call_t(){
		isApiCall=false;
	}	
	Call_t(duint rva_, duint addr_value){
		RVA = rva_;
		RelativeAddressValue = addr_value;
		isApiCall = false;
	}

	bool operator==(const Call_t& rhs) const
	{
		return (RVA == rhs.RVA);
	}
	bool operator<(const Call_t& rhs) const
	{
		return RVA < rhs.RVA;
	}

};
